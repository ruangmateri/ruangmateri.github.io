[tex]256 = \frac{n}{2} (2a + (n - 1)b) \\ 256 = \frac{n}{2} (2.4 + (n - 1)8) \\ 256 = \frac{n}{2} (8 + 8n - 8) \\ 256.2 = n(8n) \\ 512 = 8 {n}^{2} \\ \frac{512}{8} = {n}^{2} \\ 64 = {n}^{2} \\ 8 = n[/tex]
